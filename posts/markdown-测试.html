// C 语言代码示例（语法高亮）
#include <stdio.h>
#include <stdlib.h>

// 函数声明：计算两数乘积
int multiply(int a, int b);

int main() {
    int num1 = 8, num2 = 9;
    int result = multiply(num1, num2);
    
    // 输出结果
    printf("数字 %d 和 %d 的乘积是：%d\n", num1, num2, result);
    
    // 动态内存分配示例
    int *arr = (int *)malloc(3 * sizeof(int));
    if (arr != NULL) {
        arr[0] = 10;
        arr[1] = 20;
        arr[2] = 30;
        printf("数组元素：%d, %d, %d\n", arr[0], arr[1], arr[2]);
        free(arr); // 释放内存
    }
    
    return 0;
}

// 函数定义
int multiply(int a, int b) {
    return a * b;
}

# Python 代码示例（语法高亮）
def calculate_sum(a: int, b: int) -> int:
    """计算两个数的和（文档字符串）"""
    try:
        return a + b
    except TypeError as e:
        print(f"错误：{e}")
        return 0

if __name__ == "__main__":
    result = calculate_sum(5, 3)
    print(f"两数之和：{result}")

-- SQL 代码示例（创建表+查询）
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL,
    create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 查询2024年之后创建的用户，取前10条
SELECT id, username, email 
FROM users 
WHERE create_time > '2024-01-01'
ORDER BY create_time DESC
LIMIT 10;

7.1 流程图（flowchart）

flowchart TD
    A[开始] --> B{条件判断}
    B -->|是| C[执行操作1]
    B -->|否| D[执行操作2]
    C --> E[处理数据]
    D --> E
    E --> F[结束]

stateDiagram

九、特殊说明

  1. 本文档包含 Markdown 所有常用语法，复制到支持 Markdown 的编辑器（如 Typora、VS Code、在线 Markdown 编辑器）可直接渲染完整效果；

  2. 扩展语法（如高亮、任务列表、脚注）在部分编辑器（如记事本）中可能无法正常渲染，推荐使用 VS Code + Markdown All in One 插件；

  3. Mermaid 图表和 LaTeX 公式需编辑器支持（如 Typora、VS Code 需安装对应插件），在线预览可使用 Markdown Live Preview 等工具。
